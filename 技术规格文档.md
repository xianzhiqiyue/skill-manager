### 3.4 同步模式实现

```go
// internal/sync/engine.go

type SyncMode string

const (
    ModeAuto    SyncMode = "auto"      // 自动检测
    ModeSymlink SyncMode = "symlink"   // 符号链接
    ModeMirror  SyncMode = "mirror"    // 物理镜像
)

type SyncEngine struct {
    mode      SyncMode
    ideAdapters map[string]IDEAdapter
}

func (e *SyncEngine) Sync(skill *Skill, target IDE) error {
    // 1. 确定同步模式
    mode := e.resolveMode(target)

    // 2. 执行同步
    switch mode {
    case ModeSymlink:
        return e.syncSymlink(skill, target)
    case ModeMirror:
        return e.syncMirror(skill, target)
    }

    return nil
}

func (e *SyncEngine) resolveMode(target IDE) SyncMode {
    if e.mode != ModeAuto {
        return e.mode
    }

    // 检测目标 IDE 是否支持符号链接
    if target.SupportsSymlink() {
        return ModeSymlink
    }
    return ModeMirror
}
```

## 4. 注册中心 API 设计

### 4.1 路由定义

```go
// 公开 API
r.GET("/health", handlers.HealthCheck)
r.GET("/skills", handlers.ListSkills)           // 支持分页、筛选
r.GET("/skills/:namespace/:name", handlers.GetSkill)
r.GET("/skills/:namespace/:name/versions", handlers.ListVersions)
r.GET("/search", handlers.SearchSkills)          // 语义搜索

// 需要认证
auth := r.Group("/")
auth.Use(middleware.Auth())
{
    // 技能管理
    auth.POST("/skills", handlers.CreateSkill)
    auth.PUT("/skills/:namespace/:name", handlers.UpdateSkill)
    auth.DELETE("/skills/:namespace/:name", handlers.DeleteSkill)
    auth.POST("/skills/:namespace/:name/versions", handlers.PublishVersion)
    auth.DELETE("/skills/:namespace/:name/versions/:version", handlers.DeleteVersion)

    // 用户管理
    auth.GET("/user", handlers.GetCurrentUser)
    auth.GET("/user/skills", handlers.GetUserSkills)
    auth.POST("/user/api-keys", handlers.CreateAPIKey)
    auth.DELETE("/user/api-keys/:id", handlers.RevokeAPIKey)
}

// 下载（公开但限速）
r.GET("/download/:namespace/:name/:version",
    middleware.RateLimit(),
    handlers.DownloadSkill)
```

## 5. 数据库 Schema

### 5.1 PostgreSQL 表结构

```sql
-- 用户表
CREATE TABLE users (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    username        VARCHAR(32) UNIQUE NOT NULL,
    email           VARCHAR(255) UNIQUE NOT NULL,
    password_hash   VARCHAR(255),       -- 可选，支持 API Key 登录
    avatar_url      VARCHAR(500),
    is_active       BOOLEAN DEFAULT true,
    created_at      TIMESTAMPTZ DEFAULT NOW(),
    updated_at      TIMESTAMPTZ DEFAULT NOW()
);

-- API Key 表
CREATE TABLE api_keys (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id         UUID REFERENCES users(id) ON DELETE CASCADE,
    key_hash        VARCHAR(255) NOT NULL,  -- 存储 hash，不存明文
    name            VARCHAR(100),            -- 用户给 key 的命名
    last_used_at    TIMESTAMPTZ,
    expires_at      TIMESTAMPTZ,
    created_at      TIMESTAMPTZ DEFAULT NOW()
);

-- 技能表
CREATE TABLE skills (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    namespace       VARCHAR(64) NOT NULL,   -- @user 或 @org/team
    name            VARCHAR(64) NOT NULL,
    owner_id        UUID REFERENCES users(id) ON DELETE CASCADE,

    -- 元数据
    description     TEXT,
    description_zh  TEXT,
    tags            TEXT[],
    homepage        VARCHAR(500),
    license         VARCHAR(50) DEFAULT 'MIT',

    -- 统计
    download_count  BIGINT DEFAULT 0,
    rating_sum      BIGINT DEFAULT 0,       -- 总评分和
    rating_count    BIGINT DEFAULT 0,       -- 评分次数

    -- 状态
    is_public       BOOLEAN DEFAULT true,
    is_deprecated   BOOLEAN DEFAULT false,

    created_at      TIMESTAMPTZ DEFAULT NOW(),
    updated_at      TIMESTAMPTZ DEFAULT NOW(),

    UNIQUE(namespace, name)
);

-- 技能版本表
CREATE TABLE skill_versions (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    skill_id        UUID REFERENCES skills(id) ON DELETE CASCADE,
    version         VARCHAR(20) NOT NULL,   -- SemVer

    -- 元数据
    manifest        JSONB,                  -- SKILL.md 的解析结果
    dependencies    JSONB,                  -- 依赖列表

    -- 存储
    storage_path    VARCHAR(500) NOT NULL,  -- 对象存储路径
    size_bytes      BIGINT,
    checksum        VARCHAR(64),            -- SHA-256

    -- 安全审计
    scan_status     VARCHAR(20) DEFAULT 'pending', -- pending/pass/fail
    scan_result     JSONB,                  -- 扫描详情

    -- 发布信息
    published_by    UUID REFERENCES users(id),
    published_at    TIMESTAMPTZ DEFAULT NOW(),

    UNIQUE(skill_id, version)
);

-- 审计日志表
CREATE TABLE audit_logs (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id         UUID REFERENCES users(id),
    action          VARCHAR(50) NOT NULL,   -- push/pull/delete
    resource_type   VARCHAR(50) NOT NULL,   -- skill/version/user
    resource_id     UUID,
    metadata        JSONB,
    ip_address      INET,
    user_agent      TEXT,
    created_at      TIMESTAMPTZ DEFAULT NOW()
);

-- 索引
CREATE INDEX idx_skills_search ON skills USING gin(to_tsvector('english', name || ' ' || COALESCE(description, '')));
CREATE INDEX idx_skills_tags ON skills USING gin(tags);
CREATE INDEX idx_skills_owner ON skills(owner_id);
CREATE INDEX idx_versions_skill ON skill_versions(skill_id);
CREATE INDEX idx_audit_user ON audit_logs(user_id, created_at);
```

### 5.2 Redis 数据结构

```
# 会话缓存 (API Key 验证)
session:{key_hash} -> user_id (TTL: 24h)

# 搜索缓存
search:{hash(query)} -> json(results) (TTL: 5min)

# 限流计数器
ratelimit:{ip}:{endpoint} -> count (TTL: 1min)

# 技能元数据缓存
skill:{namespace}:{name} -> json(metadata) (TTL: 10min)
```

## 7. 安全策略

### 7.1 静态扫描规则

```yaml
# 危险命令黑名单
dangerous_commands:
  - pattern: "rm\s+-rf\s+/"
    severity: critical
    message: "Attempt to delete root filesystem"

  - pattern: "mkfs\.\w+"
    severity: critical
    message: "Attempt to format filesystem"

  - pattern: "dd\s+if=\S+\s+of=/dev/\w+"
    severity: critical
    message: "Direct disk write operation"

  - pattern: ":\(\)\{\s*:\|:\s*&\s*\};:"
    severity: critical
    message: "Fork bomb detected"

  - pattern: "curl\s+\S+\s*\|\s*(ba)?sh"
    severity: high
    message: "Remote code execution via pipe"

  - pattern: "wget\s+\S+\s*-O\s*-\s*\|\s*(ba)?sh"
    severity: high
    message: "Remote code execution via wget"

# 可疑代码模式
suspicious_patterns:
  - pattern: "eval\s*\("
    severity: medium
    message: "Use of eval() - potential code injection"

  - pattern: "exec\s*\("
    severity: medium
    message: "Use of exec() - ensure input is sanitized"

# Python 特定
python_dangerous:
  - pattern: "__import__\s*\(\s*['\"]os['\"]\s*\)"
    severity: medium
    message: "Dynamic import of os module"

  - pattern: "subprocess\.call\s*\([^)]*shell\s*=\s*True"
    severity: high
    message: "subprocess with shell=True"
```

### 7.2 提示词注入检测

```yaml
injection_patterns:
  - pattern: "ignore\s+(all\s+)?(previous|above|earlier)\s+instructions"
    severity: high

  - pattern: "disregard\s+(your\s+)?(system\s+)?prompt"
    severity: high

  - pattern: "DAN\s+(mode|do\s+anything\s+now)"
    severity: high

  - pattern: "jailbreak|prompt\s+injection"
    severity: medium

  - pattern: "system\s+prompt\s*:\s*"
    severity: high

  - pattern: "you\s+are\s+now\s+\w+"
    severity: low
    message: "Possible role override attempt"
```

### 7.3 扫描结果处理

```go
type ScanResult struct {
    Status      string        `json:"status"`       // pass/fail/warn
    Summary     string        `json:"summary"`
    Issues      []ScanIssue   `json:"issues"`
    ScannedAt   time.Time     `json:"scanned_at"`
}

type ScanIssue struct {
    Severity    string `json:"severity"`     // critical/high/medium/low
    Category    string `json:"category"`     // dangerous_command/injection
    File        string `json:"file"`
    Line        int    `json:"line"`
    Column      int    `json:"column"`
    Match       string `json:"match"`
    Message     string `json:"message"`
    Suggestion  string `json:"suggestion"`
}
```

**发布策略**:
- `critical`: 阻止发布，必须修复
- `high`: 阻止发布，可通过 `--force` 覆盖（需二次确认）
- `medium/low`: 警告，不影响发布

| `INVALID_REQUEST` | 400 | 请求参数错误 |
| `UNAUTHORIZED` | 401 | 未认证 |
| `FORBIDDEN` | 403 | 无权限 |
| `SKILL_NOT_FOUND` | 404 | 技能不存在 |
| `VERSION_EXISTS` | 409 | 版本已存在 |
| `VALIDATION_FAILED` | 422 | 验证失败（含扫描结果）|
| `RATE_LIMITED` | 429 | 请求过于频繁 |
| `INTERNAL_ERROR` | 500 | 服务器内部错误 |

### 10.2 文件命名规范

- 技能包: `{name}-{version}.tar.gz`
- 存储路径: `skills/{namespace}/{name}/{version}.tar.gz`
- 本地缓存: `~/.skill-home/cache/{namespace}/{name}/{version}/`

### 10.3 版本号规则

遵循 [SemVer 2.0.0](https://semver.org/lang/zh-CN/):
- `MAJOR`: 不兼容的 API 修改
- `MINOR`: 向下兼容的功能新增
- `PATCH`: 向下兼容的问题修复

预发布版本: `1.0.0-alpha.1`, `1.0.0-beta.2`
